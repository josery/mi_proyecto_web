document.addEventListener('DOMContentLoaded', function() {
    const state = {
        selectedId: null,
        currentProjectName: 'proyecto_ejemplo.json'
    };

    let taskTable;
    let ganttChart;
    let pieChart;

    // --- ELEMENTOS DEL DOM ---
    const loadButton = document.getElementById('load-button');
    const saveButton = document.getElementById('save-button');
    const saveAsButton = document.getElementById('save-as-button');
    const addActivityBtn = document.getElementById('add-activity-btn');
    const addTaskBtn = document.getElementById('add-task-btn');
    const addSubtaskBtn = document.getElementById('add-subtask-btn');
    const editTaskBtn = document.getElementById('edit-task-btn');
    const deleteTaskBtn = document.getElementById('delete-task-btn');
    const moveUpBtn = document.getElementById('move-up-btn');
    const moveDownBtn = document.getElementById('move-down-btn');
    const indentBtn = document.getElementById('indent-btn');
    const unindentBtn = document.getElementById('unindent-btn');
    const projectTitleEl = document.getElementById('project-title');
    const taskModalEl = document.getElementById('task-modal');
    const taskModal = new bootstrap.Modal(taskModalEl);
    const saveTaskBtn = document.getElementById('save-task-btn');
    const taskForm = document.getElementById('task-form');
    const modalTitle = document.getElementById('modal-title');
    const taskIdInput = document.getElementById('task-id');
    const parentIdInput = document.getElementById('parent-id');
    const taskTitleInput = document.getElementById('task-title');
    const taskResponsableInput = document.getElementById('task-responsable');
    const taskInicioInput = document.getElementById('task-inicio');
    const taskFinPrevistoInput = document.getElementById('task-fin-previsto');
    const taskFinRealInput = document.getElementById('task-fin-real');
    const taskAvanceInput = document.getElementById('task-avance');
    const ganttContainer = document.getElementById('gantt-chart');
    const pieChartCanvas = document.getElementById('pie-chart').getContext('2d');
    const tableContainer = document.getElementById("task-table");

    // --- FUNCIONES ---
    function showToast(message, type = 'info') {
        const toastContainer = document.querySelector('.toast-container');
        const toastId = `toast-${Date.now()}`;
        const toastClass = { success: 'text-bg-success', error: 'text-bg-danger', info: 'text-bg-primary', warning: 'text-bg-warning' }[type] || 'text-bg-secondary';
        const toastHTML = `<div id="${toastId}" class="toast ${toastClass}" role="alert" aria-live="assertive" aria-atomic="true"><div class="d-flex"><div class="toast-body">${message}</div><button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast" aria-label="Close"></button></div></div>`;
        toastContainer.insertAdjacentHTML('beforeend', toastHTML);
        const toastEl = document.getElementById(toastId);
        const toast = new bootstrap.Toast(toastEl, { delay: 5000 });
        toast.show();
        toastEl.addEventListener('hidden.bs.toast', () => toastEl.remove());
    }

    const fromInputDate = (d) => d ? new Date(d).toLocaleDateString('es-ES', { year: '2-digit', month: '2-digit', day: '2-digit' }) : '';
    const parseToInput = (d) => {
        if (!d) return '';
        const parts = d.split('/');
        return parts.length !== 3 ? '' : `20${parts[2]}-${parts[1]}-${parts[0]}`;
    };
    
    function recalculateTaskState(data) {
        const parse = (dateStr) => dateStr ? new Date(dateStr.split('/').reverse().join('-')) : null;
        const inicio = parse(data.inicio);
        const finPrevisto = parse(data.fin_previsto);
        const finReal = parse(data.fin_real);
        data.duracion = (inicio && finPrevisto) ? `${Math.ceil((finPrevisto - inicio) / 864e5) + 1}` : "N/D";
        data.porcentaje = finReal ? "100%" : `${parseFloat(data.avance_reportado) || 0}%`;
        data.estado = finReal ? (finPrevisto && finReal <= finPrevisto ? "COMPLETADA A TIEMPO" : "COMPLETADA CON RETRASO") : (inicio ? (new Date() > finPrevisto ? "ATRASADA" : "ACTIVA") : "PENDIENTE");
        return data;
    }

    function openTaskModalForCreate(parentId = "") {
        taskForm.reset();
        taskIdInput.value = '';
        parentIdInput.value = parentId;
        const parent = parentId ? taskTable.getRow(parentId)?.getData() : null;
        modalTitle.textContent = parent ? `Añadir a "${parent.data.titulo}"` : "Añadir Nueva Actividad";
        taskModal.show();
    }

    function openTaskModalForEdit() {
        if (!state.selectedId) return;
        const task = taskTable.getRow(state.selectedId)?.getData();
        if (!task) return;
        
        taskForm.reset();
        taskIdInput.value = task.id;
        parentIdInput.value = task.parent || '';
        modalTitle.textContent = `Editando: "${task.data.titulo}"`;
        taskTitleInput.value = task.data.titulo;
        taskResponsableInput.value = task.data.responsable;
        taskAvanceInput.value = parseFloat(task.data.avance_reportado) || 0;
        taskInicioInput.value = parseToInput(task.data.inicio);
        taskFinPrevistoInput.value = parseToInput(task.data.fin_previsto);
        taskFinRealInput.value = parseToInput(task.data.fin_real);
        taskModal.show();
    }

    function generateNextEdt(parentId) {
        const allTasks = taskTable.getData();
        if (parentId) {
            const siblings = allTasks.filter(t => t.parent === parentId);
            const maxSub = siblings.reduce((max, t) => Math.max(max, parseInt(t.id.split('.').pop() || 0)), 0);
            return `${parentId}.${maxSub + 1}`;
        }
        const maxActivity = allTasks.filter(t => !t.parent).reduce((max, t) => Math.max(max, parseInt(t.id.split('.')[0] || 0)), 0);
        return `${maxActivity + 1}.0`;
    }

    function saveTask() {
        const id = taskIdInput.value;
        const title = taskTitleInput.value;
        if (!title) return showToast("El título es obligatorio.", "warning");
        
        let taskData = {
            titulo: title,
            responsable: taskResponsableInput.value,
            avance_reportado: `${taskAvanceInput.value || 0}%`,
            inicio: fromInputDate(taskInicioInput.value),
            fin_previsto: fromInputDate(taskFinPrevistoInput.value),
            fin_real: fromInputDate(taskFinRealInput.value)
        };
        
        if (id) {
            taskTable.updateRow(id, { data: recalculateTaskState({ ...taskTable.getRow(id).getData().data, ...taskData }) });
        } else {
            const parentId = parentIdInput.value;
            const newId = generateNextEdt(parentId);
            taskData.edt = newId;
            const newRowData = { id: newId, parent: parentId, data: recalculateTaskState(taskData) };
            taskTable.addRow(newRowData, undefined, parentId || undefined);
        }
        
        taskModal.hide();
        renderSecondaryComponents();
    }

    function deleteTask() {
        if (state.selectedId && confirm(`¿Seguro que quieres eliminar la tarea seleccionada?`)) {
            taskTable.deleteRow(state.selectedId);
            updateButtonStates(null);
        }
    }

    function moveTask(dir) {
        if (!state.selectedId) return;
        const row = taskTable.getRow(state.selectedId);
        const sibling = dir === 'up' ? row.getPrevRow() : row.getNextRow();
        if (sibling) {
            row.move(sibling, dir === 'down');
            renderSecondaryComponents();
        }
    }

    function indentTask() {
        if (!state.selectedId) return;
        const row = taskTable.getRow(state.selectedId);
        const prev = row.getPrevRow();
        if (prev) {
            row.move(prev, "last");
            renderSecondaryComponents();
        }
    }

    function unindentTask() {
        if (!state.selectedId) return;
        const row = taskTable.getRow(state.selectedId);
        const parent = row.getParentRow();
        if (parent) {
            row.move(parent, "after");
            renderSecondaryComponents();
        }
    }

    function updateButtonStates(id) {
        if (state.selectedId === id) return;
        if (state.selectedId) taskTable.deselectRow(state.selectedId);
        if (id) taskTable.selectRow(id);

        state.selectedId = id;
        const selected = !!id;
        
        editTaskBtn.disabled = !selected;
        deleteTaskBtn.disabled = !selected;
        
        addActivityBtn.style.display = 'none';
        addTaskBtn.style.display = 'none';
        addSubtaskBtn.style.display = 'none';
        
        if (!selected) {
            addActivityBtn.style.display = 'inline-block';
            [moveUpBtn, moveDownBtn, indentBtn, unindentBtn].forEach(b => b.disabled = true);
            return;
        }
        
        const row = taskTable.getRow(id);
        if (!row) return;

        const hasParent = !!row.getData().parent;
        
        moveUpBtn.disabled = !row.getPrevRow();
        moveDownBtn.disabled = !row.getNextRow();
        indentBtn.disabled = !row.getPrevRow();
        unindentBtn.disabled = !hasParent;

        const edt = row.getData().data.edt || "";
        if (edt.endsWith('.0')) {
            addTaskBtn.style.display = 'inline-block';
        } else {
            addSubtaskBtn.style.display = 'inline-block';
        }
    }

    async function saveProjectData(isSaveAs = false) {
        let filename = state.currentProjectName;
        if (isSaveAs) {
            const newName = prompt("Introduce el nuevo nombre del archivo:", filename);
            if (!newName) return;
            filename = newName.endsWith('.json') ? newName : `${newName}.json`;
        }
        try {
            const data = {
                projectName: projectTitleEl.textContent.replace('AgilePlanner: ', ''),
                tasks: taskTable.getData("active")
            };
            const res = await fetch('/api/project/save', { 
                method: 'POST', 
                headers: { 'Content-Type': 'application/json' }, 
                body: JSON.stringify({ filename, projectData: data }) 
            });
            const result = await res.json();
            if (!res.ok) throw new Error(result.error);
            
            state.currentProjectName = result.filename;
            projectTitleEl.textContent = `AgilePlanner: ${result.filename.replace('.json', '')}`;
            showToast(`Proyecto guardado en: ${result.filepath}`, "success");

        } catch (e) { 
            showToast(`Error al guardar: ${e.message}`, "error"); 
        }
    }
    
    async function loadProjectData(filename) {
        if (!filename) return;
        updateButtonStates(null);
        const url = `/api/project/load/${filename}`;
        try {
            const res = await fetch(url);
            if (!res.ok) throw new Error(`Error HTTP: ${res.status}`);
            const data = await res.json();
            state.currentProjectName = filename;
            projectTitleEl.textContent = `AgilePlanner: ${data.projectName || filename.replace('.json', '')}`;
            initializeTaskTable(data.tasks || []);
        } catch (e) { 
            showToast(`Error al cargar: ${e.message}`, "error");
            initializeTaskTable([]);
        }
    }

    function renderSecondaryComponents() {
        if (!taskTable) return;
        const data = taskTable.getData("active");
        renderGantt(data);
        renderPieChart(data);
    }

    function initializeTaskTable(tasks) {
        if (taskTable) taskTable.destroy();
        
        taskTable = new Tabulator(tableContainer, {
            data: tasks,
            index: "id",
            dataTree: true,
            dataTreeStartExpanded: true,
            dataTreeElementColumn: "data.titulo",
            selectableRows: 1, 
            layout: "fitColumns",
            columns: [
                { title: "Título", field: "data.titulo", width: 300, headerSort: false },
                { title: "EDT", field: "data.edt", width: 80, headerSort: false },
                { title: "Responsable", field: "data.responsable", width: 150, headerSort: false },
                { title: "Inicio", field: "data.inicio", hozAlign: "center", headerSort: false },
                { title: "Fin Previsto", field: "data.fin_previsto", hozAlign: "center", headerSort: false },
                { title: "Fin Real", field: "data.fin_real", hozAlign: "center", headerSort: false },
                { title: "Duración", field: "data.duracion", hozAlign: "center", headerSort: false },
                { title: "% Avance", field: "data.porcentaje", hozAlign: "center", formatter: "progress", formatterParams: { legend: true }, headerSort: false },
                { title: "Estado", field: "data.estado", headerSort: false },
            ],
            dataLoaded: () => renderSecondaryComponents(),
            rowDeleted: () => {
                updateButtonStates(null);
                renderSecondaryComponents();
            },
            rowClick: function(e, row){
                updateButtonStates(row.getData().id);
            },
        });
        updateButtonStates(null);
    }
    
    function renderGantt(tasks) {
        ganttContainer.innerHTML = '';
        if (!tasks || !tasks.length) return;
        const ganttTasks = tasks.map(t => ({
            id: t.id,
            name: t.data.titulo,
            start: t.data.inicio ? new Date(t.data.inicio.split('/').reverse().join('-')) : null,
            end: t.data.fin_previsto ? new Date(t.data.fin_previsto.split('/').reverse().join('-')) : null,
            progress: parseFloat(t.data.porcentaje) || 0,
        })).filter(t => t.start && t.end);
        if (ganttTasks.length > 0) new Gantt("#gantt-chart", ganttTasks, { language: 'es' });
    }

    function renderPieChart(tasks) {
        if (pieChart) pieChart.destroy();
        const topLevel = tasks.filter(t => !t.parent);
        pieChart = new Chart(pieChartCanvas, {
            type: 'pie',
            data: {
                labels: topLevel.map(t => t.data.titulo),
                datasets: [{ data: topLevel.map(t => parseFloat(t.data.porcentaje) || 0) }]
            },
            options: { plugins: { title: { display: true, text: 'Avance por Actividad Principal' } } }
        });
    }

    tableContainer.addEventListener('click', (e) => {
        const rowEl = e.target.closest('.tabulator-row');
        if (rowEl) {
            const row = taskTable.getRow(rowEl);
            updateButtonStates(row.getData().id);
        } else {
            updateButtonStates(null);
        }
    });

    loadButton.addEventListener('click', () => {
        const f = prompt("Nombre del archivo a cargar:", state.currentProjectName);
        if (f) loadProjectData(f);
    });
    saveButton.addEventListener('click', () => saveProjectData(false));
    saveAsButton.addEventListener('click', () => saveProjectData(true));
    addActivityBtn.addEventListener('click', () => openTaskModalForCreate(""));
    addTaskBtn.addEventListener('click', () => openTaskModalForCreate(state.selectedId));
    addSubtaskBtn.addEventListener('click', () => openTaskModalForCreate(state.selectedId));
    editTaskBtn.addEventListener('click', openTaskModalForEdit);
    deleteTaskBtn.addEventListener('click', deleteTask);
    moveUpBtn.addEventListener('click', () => moveTask('up'));
    moveDownBtn.addEventListener('click', () => moveTask('down'));
    indentBtn.addEventListener('click', indentTask);
    unindentBtn.addEventListener('click', unindentTask);
    saveTaskBtn.addEventListener('click', saveTask);

    loadProjectData(state.currentProjectName);
});
